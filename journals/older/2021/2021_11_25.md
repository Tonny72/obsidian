	  }

	  - DateTime x;
	  - Console.WriteLine("Foreach");
	  sw.Restart();
	  foreach (var item in list)
	  {
	  x = item;
	  }
	  Console.WriteLine("Elapsed={0}", sw.Elapsed);
	  - Console.WriteLine("ForNext");
	  sw.Restart();
	  for (int i = 0; i < list.Count; i++)
	  {
	  x = list[i];
	  }
	  - Console.WriteLine("Elapsed={0}", sw.Elapsed);
