	  import polars as pl
	  import datetime as datetime
	  import numpy as np
	  df = pl.datetime_range(start=datetime.datetime(2018, 1, 1), end=datetime.datetime(2034,1,1), interval='1m', eager=True).alias("date").to_frame()
	  for col in range(300):
	      df = df.with_columns(pl.lit(np.random.rand(df.height)).alias(f"random{col}"))
