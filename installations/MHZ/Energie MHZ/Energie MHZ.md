- # Welke velden moet worden gelogd?
- ## SQL van MHZ maandrapport voor Ivo
  SELECT TOP (100) PERCENT YEAR(PC.timestamp) AS Jaar, MONTH(PC.timestamp) AS Maand,
  SUM(PC.PWR_TOTAL_CONSUMPTION_ZL_UV + PC.PWR_TOTAL_CONSUMPTION_PM_B1 + PC.PWR_TOTAL_CONSUMPTION_PM_B2 + PC.PWR_TOTAL_CONSUMPTION_PM_B3)
  AS \[ZuigerLimburg + Boosters\],
  SUM(PC.PWR_TOTAL_CONSUMPTION_BC101 + PC.PWR_TOTAL_CONSUMPTION_PU101 + PC.PWR_TOTAL_CONSUMPTION_VS101)
  AS \[Aanvoer zand\],
  SUM(PC.PWR_TOTAL_CONSUMPTION_AT201a + PC.PWR_TOTAL_CONSUMPTION_AT201B + PC.PWR_TOTAL_CONSUMPTION_PU201 + PC.PWR_TOTAL_CONSUMPTION_PU205
  \+ PC.PWR_TOTAL_CONSUMPTION_VS201 + PC.PWR_TOTAL_CONSUMPTION_WPU201 + PC.PWR_TOTAL_CONSUMPTION_WPU202 + PC.PWR_TOTAL_CONSUMPTION_WPU203
  \+ PC.PWR_TOTAL_CONSUMPTION_WPU204 + PC.PWR_TOTAL_CONSUMPTION_WPU205 + PC.PWR_TOTAL_CONSUMPTION_WPU206 + PC.PWR_TOTAL_CONSUMPTION_WPU501
  \+ PC.PWR_TOTAL_CONSUMPTION_WPU502 + PC.PWR_TOTAL_CONSUMPTION_BL401 + PC.PWR_TOTAL_CONSUMPTION_CN401 + PC.PWR_TOTAL_CONSUMPTION_CN402
  \+ PC.PWR_TOTAL_CONSUMPTION_FL401 + PC.PWR_TOTAL_CONSUMPTION_FL402 + PC.PWR_TOTAL_CONSUMPTION_FL403 + PC.PWR_TOTAL_CONSUMPTION_FL404
  \+ PC.PWR_TOTAL_CONSUMPTION_PU401 + PC.PWR_TOTAL_CONSUMPTION_PU402 + PC.PWR_TOTAL_CONSUMPTION_PU403 + PC.PWR_TOTAL_CONSUMPTION_VS401
  \+ PC.PWR_TOTAL_CONSUMPTION_VS402 +
  IIF( PC.PWR_TOTAL_CONSUMPTION_CPU401 IS NULL, 0, PC.PWR_TOTAL_CONSUMPTION_CPU401) + PC.PWR_TOTAL_CONSUMPTION_WPU401 + PC.PWR_TOTAL_CONSUMPTION_WPU402
  \+ PC.PWR_TOTAL_CONSUMPTION_MI401 + PC.PWR_TOTAL_CONSUMPTION_WPU104 + PC.PWR_TOTAL_CONSUMPTION_WPU601 + PC.PWR_TOTAL_CONSUMPTION_WPU604)
  AS Wasserij,
  SUM(PC.PWR_TOTAL_CONSUMPTION_FP651 + PC.PWR_TOTAL_CONSUMPTION_FP652 + PC.PWR_TOTAL_CONSUMPTION_WPU651 + PC.PWR_TOTAL_CONSUMPTION_MI651
  \+ PC.PWR_TOTAL_CONSUMPTION_MI653 + PC.PWR_TOTAL_CONSUMPTION_SC651 + PC.PWR_TOTAL_CONSUMPTION_WPU602 + PC.PWR_TOTAL_CONSUMPTION_TH601
  \+ PC.PWR_TOTAL_CONSUMPTION_PU603 + PC.PWR_TOTAL_CONSUMPTION_WPU606A + PC.PWR_TOTAL_CONSUMPTION_WPU606B + PC.PWR_TOTAL_CONSUMPTION_MI601A
  \+ PC.PWR_TOTAL_CONSUMPTION_MI601B + PC.PWR_TOTAL_CONSUMPTION_MI602 + PC.PWR_TOTAL_CONSUMPTION_CPU601 + PC.PWR_TOTAL_CONSUMPTION_CPU602)
  AS Waterbehandeling,
  SUM(PC.InkAfn_PWR_T1 + PC.InkAfn_PWR_T2) AS InkAfn_kWh,
  SUM(PC.InkInj_PWR_T1 + PC.InkInj_PWR_T2) AS InkInj_kWh,
  SUM(PC.GS_Prd_PWR_T1 + PC.GS_Prd_PWR_T2) AS GS_Prd_kWh, SUM(PC.InkAfn_PWR_T1) AS Afname_Dag, SUM(PC.InkAfn_PWR_T2) AS Afname_Nacht
  FROM dbo.LNKWRX_DATALOG_TM_0D0H1M0S_DIFFERENCES AS PC LEFT OUTER JOIN
  dbo.LNKWRX_DATALOG_TM_0D0H5M0S_DIFFERENCES AS M ON PC.timestamp = M.timestamp
  GROUP BY YEAR(PC.timestamp), MONTH(PC.timestamp)
  ORDER BY Jaar, Maand
